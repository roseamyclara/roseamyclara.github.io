function attemptLogin() { // 使用 .trim() 自動去掉使用者不小心輸入的前後空格 const enteredPassword = passwordInput.value.trim(); if (enteredPassword === CORRECT_PASSWORD) { loginContainer.style.display = 'none'; if(loginError) loginError.textContent = ''; document.body.classList.remove('login-prompt'); document.body.classList.add('app-loaded'); document.title = '編制外教師'; appWrapper.style.display = 'block'; initializeAppInterface(); } else { if(loginError) { // 提示更具體一點，幫助檢查 loginError.textContent = '密碼錯誤。請注意大小寫及是否為半形字元。'; } passwordInput.value = ''; passwordInput.focus(); } }
